const initialState = { name: '', email: '', password: '', loading: false, error: null };

function reducer(state, action) {
  switch (action.type) {
    case 'setField':
      return { ...state, [action.field]: action.value };
    case 'submit':
      return { ...state, loading: true, error: null };
    case 'success':
      return { ...initialState }; // reset form
    case 'error':
      return { ...state, loading: false, error: action.message };
  }
}

function SignupForm() {
  const [state, dispatch] = useReducer(reducer, initialState);

  const handleSubmit = async () => {
    dispatch({ type: 'submit' });
    try {
      await api.signup(state);
      dispatch({ type: 'success' });
    } catch (e) {
      dispatch({ type: 'error', message: e.message });
    }
  };

  return (
    <form>
      <input value={state.name} onChange={e => dispatch({ type: 'setField', field: 'name', value: e.target.value })} />
      <input value={state.email} onChange={e => dispatch({ type: 'setField', field: 'email', value: e.target.value })} />
      {state.error && <p>{state.error}</p>}
      <button onClick={handleSubmit} disabled={state.loading}>
        {state.loading ? 'Saving...' : 'Sign up'}
      </button>
    </form>
  );
}
Problem solved: State transitions that need to be atomic. When you submit, you want loading: true AND error: null to happen together in one render — not two separate setState calls that could cause two renders or a flash of stale error message.

Example 2: Shopping cart


function reducer(state, action) {
  switch (action.type) {
    case 'addItem': {
      const exists = state.items.find(i => i.id === action.item.id);
      if (exists) {
        return {
          ...state,
          items: state.items.map(i =>
            i.id === action.item.id ? { ...i, qty: i.qty + 1 } : i
          )
        };
      }
      return { ...state, items: [...state.items, { ...action.item, qty: 1 }] };
    }
    case 'removeItem':
      return { ...state, items: state.items.filter(i => i.id !== action.id) };
    case 'clearCart':
      return { ...state, items: [] };
  }
}
Problem solved: Business logic lives in the reducer, not scattered across components. Any component that dispatches addItem gets the same behavior — the "if it already exists, increment qty" logic is in one place.

Example 3: Multi-step wizard / checkout flow


const initialState = { step: 1, shippingInfo: null, paymentInfo: null, orderId: null };

function reducer(state, action) {
  switch (action.type) {
    case 'nextStep':
      return { ...state, step: state.step + 1 };
    case 'prevStep':
      return { ...state, step: state.step - 1 };
    case 'saveShipping':
      return { ...state, shippingInfo: action.data, step: 2 };
    case 'savePayment':
      return { ...state, paymentInfo: action.data, step: 3 };
    case 'orderPlaced':
      return { ...state, orderId: action.orderId, step: 4 };
  }
}
Problem solved: The entire flow state is in one place. Any step component only dispatches actions — it doesn't need to know what step comes next or what other steps have stored. You can also serialize the whole state and restore a user's progress.

The pattern behind all three examples:

Problem	Why useReducer wins
Multiple values that must change atomically	One dispatch = one render
Business logic growing inside components	Reducer is pure, testable in isolation
State transitions that depend on previous state	Reducer always receives the current state
Debugging — "how did we get into this state?"	You can log every action and replay them
That last point is big — because actions are plain objects, you can log every dispatch and build a complete history of what happened. Redux DevTools works on exactly this principle.

Question 13: useMemo and useCallback
These two hooks solve the same core problem: React re-renders components on every state/prop change, and sometimes that's expensive.

The problem first

function ProductList({ products, onSelect }) {
  const filtered = products.filter(p => p.inStock); // runs on every render
  return filtered.map(p => <Product key={p.id} product={p} onClick={onSelect} />);
}
If the parent re-renders for an unrelated reason (e.g. a clock ticking), ProductList re-renders too, and that .filter() runs again even if products didn't change.

useMemo — memoize a computed value

function ProductList({ products, onSelect }) {
  const filtered = useMemo(
    () => products.filter(p => p.inStock),
    [products]  // only recompute when products changes
  );
  return filtered.map(p => <Product key={p.id} product={p} onClick={onSelect} />);
}
Think of it like Node's module-level caching pattern — the result is cached and only recomputed when the dependency changes.

When to use: expensive calculations (sorting/filtering large arrays, derived data, heavy transforms).

useCallback — memoize a function reference

function ProductList({ products, onSelect }) {
  const handleSelect = useCallback(
    (id) => onSelect(id),
    [onSelect]  // only recreate when onSelect changes
  );
  return filtered.map(p => <Product key={p.id} product={p} onClick={handleSelect} />);
}
Why functions need this: In JS, () => {} creates a new object on every call. So even onClick={() => doSomething(id)} is a different reference every render — which breaks child component memoization.

The dependency array — same as useEffect
Both take [deps] as the second argument, same rules as useEffect:

Empty [] → compute/create once, never again
[a, b] → recompute when a or b changes
No array → defeats the purpose entirely
The pair that makes it actually work: React.memo
useMemo/useCallback only pay off when the child component is wrapped in React.memo:


// Without React.memo — child re-renders regardless of stable props
const Product = ({ product, onClick }) => <div onClick={() => onClick(product.id)}>{product.name}</div>;

// With React.memo — child only re-renders if props actually changed
const Product = React.memo(({ product, onClick }) => (
  <div onClick={() => onClick(product.id)}>{product.name}</div>
));
React.memo is the equivalent of shouldComponentUpdate — it does a shallow comparison of props and skips the render if nothing changed. But if you pass a new function reference on every render, React.memo sees it as a changed prop and re-renders anyway — that's why useCallback exists.

The full pattern together

function App() {
  const [products, setProducts] = useState([...]);
  const [cart, setCart] = useState([]);

  // Stable function reference — only changes if setCart changes (it never does)
  const handleAddToCart = useCallback((id) => {
    setCart(prev => [...prev, id]);
  }, []);

  // Expensive computation — only reruns when products changes
  const inStock = useMemo(
    () => products.filter(p => p.inStock).sort((a, b) => a.price - b.price),
    [products]
  );

  return <ProductList products={inStock} onAdd={handleAddToCart} />;
}

const ProductList = React.memo(({ products, onAdd }) => {
  // This only re-renders when products or onAdd actually changes
  return products.map(p => <Product key={p.id} product={p} onAdd={onAdd} />);
});
The honest caveat
Don't reach for these by default. Premature optimization here adds complexity for no gain. The rule of thumb:

Situation	Use it?
List with 10 items, simple render	No
List with 1000+ items or heavy DOM	Yes
Function passed to React.memo child	Yes
Function used inside useEffect deps	Yes (to avoid infinite loops)
Everything, just in case	No — it has overhead too
The React team's own guidance: profile first, optimize second.

Question 15: React Router
React Router is the standard way to handle navigation in React apps. Think of it like an Express router — but instead of mapping URLs to handler functions on the server, it maps URLs to components on the client.

Setup

npm install react-router-dom
The core concept: declarative routing
Express style (imperative — you define handlers):


app.get('/users/:id', (req, res) => res.send(getUser(req.params.id)));
React Router style (declarative — you define what renders):


<Route path="/users/:id" element={<UserProfile />} />
Same idea, different execution: the URL determines what component is mounted.

Basic setup

import { BrowserRouter, Routes, Route, Link } from 'react-router-dom';

function App() {
  return (
    <BrowserRouter>
      <nav>
        <Link to="/">Home</Link>
        <Link to="/products">Products</Link>
        <Link to="/about">About</Link>
      </nav>

      <Routes>
        <Route path="/" element={<Home />} />
        <Route path="/products" element={<Products />} />
        <Route path="/about" element={<About />} />
        <Route path="*" element={<NotFound />} />  {/* catch-all */}
      </Routes>
    </BrowserRouter>
  );
}
BrowserRouter — wraps the app, listens to window.history
Routes — renders the first Route that matches (like a switch statement)
Route — maps a path to a component
Link — renders an <a> tag that doesn't do a full page reload
URL parameters — same as Express :param

<Route path="/products/:id" element={<ProductDetail />} />

import { useParams } from 'react-router-dom';

function ProductDetail() {
  const { id } = useParams();  // same as req.params.id in Express

  const [product, setProduct] = useState(null);

  useEffect(() => {
    fetch(`/api/products/${id}`).then(r => r.json()).then(setProduct);
  }, [id]);

  if (!product) return <p>Loading...</p>;
  return <h1>{product.name}</h1>;
}

Nested routes — layouts with shared UI
This is where React Router gets powerful. Say you want /dashboard, /dashboard/settings, and /dashboard/profile to all share a sidebar:


<Routes>
  <Route path="/dashboard" element={<DashboardLayout />}>
    <Route index element={<DashboardHome />} />         {/* /dashboard */}
    <Route path="settings" element={<Settings />} />    {/* /dashboard/settings */}
    <Route path="profile" element={<Profile />} />      {/* /dashboard/profile */}
  </Route>
</Routes>
The parent layout uses <Outlet /> to say "render the child route here":


import { Outlet, Link } from 'react-router-dom';

function DashboardLayout() {
  return (
    <div style={{ display: 'flex' }}>
      <aside>
        <Link to="/dashboard">Home</Link>
        <Link to="/dashboard/settings">Settings</Link>
        <Link to="/dashboard/profile">Profile</Link>
      </aside>
      <main>
        <Outlet />  {/* child route renders here */}
      </main>
    </div>
  );
}
Outlet is like {children} but driven by the URL instead of props.

Protected routes — guard pattern
A common pattern: redirect unauthenticated users away from private pages.


function ProtectedRoute({ children }) {
  const { user } = useAuth();  // your auth context
  
  if (!user) {
    return <Navigate to="/login" replace />;
  }
  
  return children;
}
